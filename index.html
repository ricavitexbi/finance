Database Structure

Important: If you're getting 'user_id' errors, use the SIMPLE structure below:
Option 1 - SIMPLE Structure (recommended to start):
CREATE TABLE transactions (
    id SERIAL PRIMARY KEY,
    data DATE NOT NULL,
    tipo VARCHAR(10) NOT NULL,
    categoria VARCHAR(50) NOT NULL,
    valor DECIMAL(10,2) NOT NULL,
    descricao TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Optional: Create index for better performance
CREATE INDEX idx_transactions_data ON transactions(data DESC);
Option 2 - COMPLETE Structure (with user isolation):
CREATE TABLE transactions (
    id SERIAL PRIMARY KEY,
    user_id UUID REFERENCES auth.users(id),
    data DATE NOT NULL,
    tipo VARCHAR(10) NOT NULL,
    categoria VARCHAR(50) NOT NULL,
    valor DECIMAL(10,2) NOT NULL,
    descricao TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Enable RLS
ALTER TABLE transactions ENABLE ROW LEVEL SECURITY;

-- Create policy
CREATE POLICY "Users can only see own transactions" 
ON transactions FOR ALL 
USING (auth.uid() = user_id);
